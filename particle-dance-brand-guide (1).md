  text-shadow:

    0 0 20px rgba(255, 255, 255, 0.2),
    0 0 40px rgba(255, 255, 255, 0.1);

.text-glow-color {
  text-shadow: 
    0 0 10px var(--particle-color),
    0 0 20px var(--particle-color),
    0 0 40px var(--particle-color);
